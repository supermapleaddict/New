%This project basically allows you to input any number between 100-150 and get the amount seperated into 50s/10s/1s bills.%
prompt = 'Hello! Welcome, to start, please put the amount you want (100-150) into cashout';

x = input(prompt)
Cashout= x

for i=Cashout
   if i/50 ==2
       R4= [2,0,0];
       R1= 0;

   else
       R1=i/50 -2;
   end
   if i/50 ==3
       R4= [3,0,0];
           R1= 0;

   end

end

for d=R1
    if (d<.2)&& (5*d>.01)
        R4= [2,0,50*d];

    end
    if (.2<d) && (5*d<5)
        R3= 5*R1 ;
    else
        R3=0;
    end

    if (0.2<d) && (5*d<1.00001) %An example of how imporant domain and how numbers can be wrong; (5*d>1.)
           R4= [2,1,0];

    end
    if (1.0001<5*d) && (5*d<2)
           R4= [2,2,0];

    end
    if (3<5*d) && (5*d<3.9999) %The .9999 here is due to the variable 5*d being less than 4. mat lab couldn't distinguish it correctly when using "4"%
           R4= [2,3,0];

    end
    if (3.99999<5*d) && (5*d<5)%Same from above%
           R4= [2,4,0];

    end
end

for f= R3
    if  (5*(f-1)>0.001) && ((f-1)<.9999) %The domain is specific here because f doesn't equal a whole number. so when subtracting, one might get 8.2*e-64.
        R4= [2,1,10*(R3-1)];             %The function is 10(R3-1) because R3 is already multiplied by 5; similar to the prievious function from before
    end                                  %in the tens paragraph, (50*d), there is a difference however. The formula is (R3-n) where n is the 10s place
    if ((f-2)<.9999) &&(5*(f-2)>0.001)   %Without the function, the entire ones place would be +10, +20, +n*10 more than what it should be.
        R4= [2,2,10*(R3-2)];
    end
    if ((f-3)<.9999) &&(5*(f-3)>0.001)
        R4= [2,3,10*(R3-3)];
    end
    if ((f-4)<.9999) &&(5*(f-4)>0.001)
        R4= [2,4,10*(R3-4)];
    end
end

X5=[R4(1,end), "1$ bills"];
X3=[R4(1,2), "10$ bills"];
